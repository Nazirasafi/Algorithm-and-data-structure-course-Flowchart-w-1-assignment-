class Node:
    def __init__(self, name, age, gender):
        self.name = name
        self.age = age
        self.gender = gender
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def add_front(self, new_node):
        if self.head is None:
            self.head = new_node
        else:
            new_node.next = self.head
            self.head = new_node

    def add_end(self, new_node):
        if self.head is None:
            self.head = new_node
        else:
            tmp = self.head
            while tmp.next is not None:
                tmp = tmp.next
            tmp.next = new_node

    def add_between(self, prev_node, new_node):
        if prev_node is not None:
            new_node.next = prev_node.next
            prev_node.next = new_node

    def display(self):
        tmp = self.head
        while tmp:
            print(f"[{tmp.name}]", end=" -> ")
            tmp = tmp.next
        print("None")
